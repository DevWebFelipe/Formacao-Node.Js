iniciar o npm        -> 'npm init -y'
instalar o ts        -> 'npm i -D typescript'
instalar o ts-node   -> 'npm install -D ts-node'
arquivo de config ts -> 'npx tsc --init'
fastify              -> 'npm i fastify'
ler ts com node      -> 'npm install -D @types/node'
converter ts para js -> 'npm install tsx -D'
eslint da rocket     -> 'npm i eslint @rocketseat/eslint-config -D'
instalar knex        -> 'npm i knex'
iniciar knex         -> 'npx knex init'
instalar dive do db  -> 'npm install knex sqlite3'
instalar dotEnv      -> 'npm i dotenv'
validacao de dados   -> 'npm i zod'
gerenciando cookies  -> 'npm i @fastify/cookie'
vitest para teste    -> 'npm i vitest -D'
servidor teste node  -> 'npm i supertest -D'
ts supertest         -> 'npm i --save-dev @types/supertest'

criar migration      -> 'npm run knex -- migrate:make nome-migrate -x ts'
executar ultimo mig  -> 'npm run knex -- migrate:latest'
reverter ultima mig  -> 'npm run knex -- migrate:rollback'
adicionar algo a tab -> 'npm run knex -- migrate:make add-session-id-to-transactions'
